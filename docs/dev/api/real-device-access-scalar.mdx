import BrowserOnly from '@docusaurus/BrowserOnly';
import useBaseUrl from '@docusaurus/useBaseUrl';
import '@scalar/api-reference-react/style.css';

export const ApiReference = () => {
  // useBaseUrl prefixes the site baseUrl so the spec resolves correctly under
  // PR-preview deploys (where BASE_URL is set) as well as locally/production.
  const specUrl = useBaseUrl('/oas/real-device-access-api.yaml');
  return (
    <BrowserOnly fallback={<div>Loading API reference…</div>}>
      {() => {
        const { ApiReferenceReact } = require('@scalar/api-reference-react');
        return (
          <ApiReferenceReact
            configuration={{
              url: specUrl,
              layout: 'classic',
              showSidebar: true,
              defaultOpenAllTags: true,
              hideClientButton: true,
              hideTestRequestButton: true,
              hideSearch: true,
            }}
          />
        );
      }}
    </BrowserOnly>
  );
};
