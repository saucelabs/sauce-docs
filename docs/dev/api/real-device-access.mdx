import Head from '@docusaurus/Head';
import BrowserOnly from '@docusaurus/BrowserOnly';
import useBaseUrl from '@docusaurus/useBaseUrl';

{/* Tags <body> with a class so the full-width API-reference styling in
    custom.css applies only to these embedded Scalar pages. */}

export const ApiReference = () => {

  const specUrl = useBaseUrl('/oas/real-device-access-api-spec.yaml');
  return (
    <BrowserOnly fallback={<div>Loading API reference…</div>}>
      {() => {
        const { ApiReferenceReact } = require('@scalar/api-reference-react');
        return (
          <ApiReferenceReact
            configuration={{
              sources: [
                {
                  title: 'Real Device Access API',
                  url: specUrl,
                },
              ],
              layout: 'classic',
              showSidebar: true,
              defaultOpenAllTags: true,
              hideClientButton: true,
              hideTestRequestButton: true,
              hideSearch: true,
              hideDarkModeToggle: true,
            }}
          />
        );
      }}
    </BrowserOnly>
  );
};

<ApiReference />
